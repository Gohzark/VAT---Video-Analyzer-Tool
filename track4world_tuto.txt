# 1. Patch PyTorch pour ignorer le mismatch CUDA 13.2 vs 12.8
sed -i '479s/raise RuntimeError(CUDA_MISMATCH_MESSAGE.format(cuda_str_version, torch.version.cuda))/warnings.warn(CUDA_MISMATCH_MESSAGE.format(cuda_str_version, torch.version.cuda))/' \
    /home/tino/apps/miniconda3/envs/track4world/lib/python3.11/site-packages/torch/utils/cpp_extension.py

# 2. Patch setup.py de GroundingDINO pour retirer les archis non supportées par CUDA 13.2
sed -i '/gencode=arch=compute_70/d' grounding_dino/setup.py
sed -i '/gencode=arch=compute_75/d' grounding_dino/setup.py

# 3. Variables d'environnement
